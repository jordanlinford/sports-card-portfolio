import { TakeInputs } from "./types";

// Replace `any` with your actual MarketOutlook type.
export function marketToTakeInputs(market: any): TakeInputs {
  // 1) Choose ONE existing “how much data do we have?” field already in your app.
  // Examples: market.hitCount, market.sampleSize, market.marketCount, market.totalResults, etc.
  const signalCount =
    market?.signalCount ??
    market?.marketCount ??
    market?.totalCount ??
    market?.sampleSize ??
    market?.hits ??
    0;

  // 2) Pull narrative from what you already display (plain-language summary, flags, etc.)
  const narrativeText =
    (market?.narrativeSummary ??
      market?.summary ??
      market?.plainEnglish ??
      market?.insight ??
      "") as string;

  // 3) Derive tags using simple keyword rules (Phase 1). This is deterministic and stable.
  const t = narrativeText.toLowerCase();
  const narrativeTags: string[] = [];

  if (t.includes("rookie")) narrativeTags.push("rookie");
  if (t.includes("breakout") || t.includes("camp buzz") || t.includes("trending up")) narrativeTags.push("breakout");
  if (t.includes("injury") || t.includes("out for season") || t.includes("questionable")) narrativeTags.push("injury");
  if (t.includes("benched") || t.includes("backup") || t.includes("demoted")) narrativeTags.push("benched");
  if (t.includes("decline") || t.includes("washed") || t.includes("lost a step")) narrativeTags.push("decline");
  if (t.includes("hype") || t.includes("viral") || t.includes("highlight")) narrativeTags.push("hype");

  // If you already have explicit flags, use them too:
  if (Array.isArray(market?.flags)) {
    for (const f of market.flags) narrativeTags.push(String(f).toLowerCase());
  }

  // Optional: if you already compute an internal confidence score, reuse it.
  const confidenceScore = market?.confidenceScore ?? market?.dataConfidence ?? undefined;

  return {
    signalCount,
    confidenceScore,
    narrativeTags: Array.from(new Set(narrativeTags)),
    sport: market?.sport,
    position: market?.position,
    age: market?.age,
    isRookie: market?.isRookie,
    seasonPhase: market?.seasonPhase,
  };
}

C. The take generator (rules + templates)
Create server/src/takes/generator.ts

import { GenerateTakesRequest, Take } from "./types";

function clamp(n: number, min: number, max: number) {
  return Math.max(min, Math.min(max, n));
}

function tierFromSignalCount(n: number) {
  // Tune these to match your app’s current “7 hits is liquid” mindset.
  // This is not eBay-specific; it’s your existing market coverage count.
  if (n < 7) return "THIN";
  if (n < 20) return "MEDIUM";
  return "STRONG";
}

function has(tags: string[], key: string) {
  return tags.map((t) => t.toLowerCase()).includes(key.toLowerCase());
}

export function generateTakes(req: GenerateTakesRequest): Take[] {
  const { scope, subject, inputs, portfolioContext } = req;

  const playerName = subject.playerName;
  const tags = inputs.narrativeTags ?? [];
  const signalTier = tierFromSignalCount(inputs.signalCount ?? 0);

  const takes: Omit<Take, "id">[] = [];

  // Gate: if signal is THIN, do not do bold marketing lines. Only caution.
  if (signalTier === "THIN") {
    takes.push({
      scope,
      severity: "SOFT",
      type: "CAUTION",
      trigger: "SIGNAL_THIN",
      confidence: 78,
      text: "Signal is thin here. Treat this as sentiment, not a conviction play.",
    });
    return takes.map((t) => ({ ...t, id: `${t.scope}:${t.trigger}:${t.type}:${t.severity}` }));
  }

  // STRUCTURAL: hobby moved on (narrative driven; no eBay dependence)
  if (scope !== "portfolio" && (has(tags, "decline") || has(tags, "benched"))) {
    const base = signalTier === "STRONG" ? 90 : 84;
    const severity = base >= 88 ? "STRONG" : "SOFT";
    takes.push({
      scope,
      severity,
      type: "STRUCTURAL",
      trigger: "HOBBY_MOVED_ON",
      confidence: base,
      text:
        severity === "STRONG"
          ? `The hobby already moved on from ${playerName ?? "this player"}. These markets usually don’t circle back.`
          : "Collectors aren’t waiting for a comeback here. The hobby has mostly moved on.",
    });
  }

  // STRUCTURAL: QB age cliff (optional, uses your existing player metadata if you have it)
  const pos = (inputs.position ?? "").toUpperCase();
  if (scope !== "portfolio" && pos === "QB" && typeof inputs.age === "number" && inputs.age >= 33 && !inputs.isRookie) {
    const base = clamp(82 + (inputs.age - 33) * 1.5 + (signalTier === "STRONG" ? 4 : 0), 75, 93);
    const severity = base >= 88 ? "STRONG" : "SOFT";
    takes.push({
      scope,
      severity,
      type: "STRUCTURAL",
      trigger: "QB_UPSIDE_WINDOW",
      confidence: base,
      text:
        severity === "STRONG"
          ? `${playerName ?? "This player"}’s card market is structurally capped. The ceiling doesn’t come back.`
          : `${playerName ?? "This player"} is past the hobby’s upside window. Even solid games rarely rebuild conviction.`,
    });
  }

  // TIMING: rookie uncertainty (marketing-usable without price-level claims)
  if (scope !== "portfolio" && inputs.isRookie) {
    const base = signalTier === "STRONG" ? 88 : 82;
    const severity = base >= 88 ? "STRONG" : "SOFT";
    takes.push({
      scope,
      severity,
      type: "TIMING",
      trigger: "ROOKIE_UNCERTAINTY_WINDOW",
      confidence: base,
      text:
        severity === "STRONG"
          ? `This is the classic rookie uncertainty window for ${playerName ?? "this player"}. Once clarity hits, the market re-prices fast.`
          : `${playerName ?? "This rookie"} is priced like uncertainty is permanent. That rarely lasts once games start.`,
    });
  }

  // CAUTION: hype without conviction
  if (scope !== "portfolio" && has(tags, "hype") && signalTier === "MEDIUM") {
    takes.push({
      scope,
      severity: "SOFT",
      type: "CAUTION",
      trigger: "HYPE_WITHOUT_CONVICTION",
      confidence: 78,
      text: "There’s a lot of attention here, but it reads like curiosity more than conviction.",
    });
  }

  // PORTFOLIO: concentration (portfolio-only, derived from portfolio context you already compute)
  if (scope === "portfolio" && portfolioContext) {
    const pct = portfolioContext.concentrationPctTop1 ?? 0;
    if (pct >= 25) {
      const conf = clamp(82 + (pct - 25) * 0.5, 70, 90);
      const severity = conf >= 88 ? "STRONG" : "SOFT";
      const label = portfolioContext.topExposureLabel ?? "one segment";
      takes.push({
        scope,
        severity,
        type: "CAUTION",
        trigger: "PORTFOLIO_CONCENTRATION",
        confidence: conf,
        text:
          severity === "STRONG"
            ? `Your portfolio is concentrated in ${label}. That’s fine when it’s hot, dangerous when it cools.`
            : `You’re overweight in ${label}. If sentiment turns, it will hit your portfolio fast.`,
      });
    }
  }

  // Rank and cap per scope
  const max = scope === "card" ? 1 : 2;
  const ranked = takes.sort((a, b) => b.confidence - a.confidence).slice(0, max);

  return ranked.map((t) => ({ ...t, id: `${t.scope}:${t.trigger}:${t.type}:${t.severity}` }));
}


D. Expose an API route that takes your existing market object
This is how you ensure you’re using market data already used in the app: the request includes the same market object you already render from.

Create server/src/routes/takes.routes.ts

import { Router } from "express";
import { generateTakes } from "../takes/generator";
import { marketToTakeInputs } from "../takes/adapter";
import { GenerateTakesRequest } from "../takes/types";

export const takesRouter = Router();

takesRouter.post("/from-market", (req, res) => {
  const { scope, subject, market, portfolioContext } = req.body as any;

  if (!scope || !subject || !market) {
    return res.status(400).json({ error: "Missing scope, subject, or market" });
  }

  const inputs = marketToTakeInputs(market);

  const payload: GenerateTakesRequest = {
    scope,
    subject,
    inputs,
    portfolioContext,
  };

  const takes = generateTakes(payload);
  res.json({ takes });
});

export async function getTakesFromMarket(payload: {
  scope: "card" | "player" | "portfolio";
  subject: { playerName?: string; cardName?: string; subjectId?: string; portfolioId?: string };
  market: any; // this is your existing market data object already being used in the app

    method: "POST",
    headers: { "Content-Type": "application/json" },

  const data = await res.json();
  return data.takes ?? [];
}

import React, { useMemo, useRef, useState } from "react";
import html2canvas from "html2canvas";

type ShareMetric = { label: string; value: string };

export function ShareCard({
  title,
  subtitle,
  takeLabel = "Collector Take",
  takeText,
  metrics = [],
  shareUrl,
  watermark = "sportscardportfolio.io",
  theme = "dark",
}: {
  title: string;                 // e.g., "Drake Maye"
  subtitle?: string;             // e.g., "2024 Prizm Silver • PSA 10" (optional)
  takeLabel?: string;            // default "Collector Take"
  takeText: string;              // the one-liner
  metrics?: ShareMetric[];       // optional: 2–4 small metrics you already have
  shareUrl?: string;             // optional deep link you copy
  watermark?: string;            // small footer text
  theme?: "dark" | "light";
}) {
  const ref = useRef<HTMLDivElement | null>(null);
  const [isDownloading, setIsDownloading] = useState(false);
  const [copied, setCopied] = useState(false);

  const bg = theme === "dark" ? "#0B0F17" : "#FFFFFF";
  const fg = theme === "dark" ? "#EAF0FF" : "#111827";
  const subtle = theme === "dark" ? "rgba(234,240,255,0.65)" : "rgba(17,24,39,0.65)";
  const border = theme === "dark" ? "rgba(234,240,255,0.12)" : "rgba(17,24,39,0.12)";
  const panel = theme === "dark" ? "rgba(234,240,255,0.06)" : "rgba(17,24,39,0.04)";

  const filename = useMemo(() => {
    const base = `${title}${subtitle ? " - " + subtitle : ""}`.replace(/[^\w\s-]/g, "").trim();
    return `${base || "share-card"}`.slice(0, 80);
  }, [title, subtitle]);

  async function onCopyLink() {
    if (!shareUrl) return;
    try {
      await navigator.clipboard.writeText(shareUrl);
      setCopied(true);
      setTimeout(() => setCopied(false), 1200);
    } catch {
      // fallback
      const el = document.createElement("textarea");
      el.value = shareUrl;
      document.body.appendChild(el);
      el.select();
      document.execCommand("copy");
      document.body.removeChild(el);
      setCopied(true);
      setTimeout(() => setCopied(false), 1200);
    }
  }

  async function onDownloadImage() {
    if (!ref.current) return;
    setIsDownloading(true);
    try {
      // scale improves sharpness for sharing
      const canvas = await html2canvas(ref.current, {
        backgroundColor: bg,
        scale: 2,
        useCORS: true,
      });
      const link = document.createElement("a");
      link.download = `${filename}.png`;
      link.href = canvas.toDataURL("image/png");
      link.click();
    } finally {
      setIsDownloading(false);
    }
  }

  async function onNativeShare() {
    if (!shareUrl) return;
    // Web Share API (works on many mobile browsers)
    // Do not rely on it; keep copy/download as primary.
    // @ts-ignore
    if (navigator.share) {
      try {
        // @ts-ignore
        await navigator.share({ title, text: takeText, url: shareUrl });
      } catch {
        // ignored
      }
    }
  }

  return (
    <div>
      <div
        ref={ref}
        style={{
          width: 640,
          maxWidth: "100%",
          borderRadius: 18,
          border: `1px solid ${border}`,
          background: bg,
          color: fg,
          padding: 18,
          boxSizing: "border-box",
        }}
      >
        <div style={{ display: "flex", justifyContent: "space-between", gap: 12 }}>
          <div style={{ minWidth: 0 }}>
            <div style={{ fontSize: 18, lineHeight: 1.2, letterSpacing: 0.2, whiteSpace: "nowrap", overflow: "hidden", textOverflow: "ellipsis" }}>
              {title}
            </div>
            {subtitle ? (
              <div style={{ marginTop: 4, fontSize: 12, color: subtle, whiteSpace: "nowrap", overflow: "hidden", textOverflow: "ellipsis" }}>
                {subtitle}
              </div>
            ) : null}
          </div>

          <div style={{ fontSize: 11, color: subtle, alignSelf: "flex-start" }}>
            {new Date().toLocaleDateString()}
          </div>
        </div>

        <div style={{ height: 12 }} />

        <div
          style={{
            borderRadius: 14,
            border: `1px solid ${border}`,
            background: panel,
            padding: 14,
          }}
        >
          <div style={{ fontSize: 11, color: subtle, marginBottom: 8 }}>{takeLabel}</div>
          <div style={{ fontSize: 15, lineHeight: 1.35 }}>{takeText}</div>
        </div>

        {metrics.length ? (
          <>
            <div style={{ height: 12 }} />
            <div style={{ display: "grid", gridTemplateColumns: metrics.length >= 4 ? "repeat(4, 1fr)" : `repeat(${metrics.length}, 1fr)`, gap: 10 }}>
              {metrics.slice(0, 4).map((m, idx) => (
                <div
                  key={idx}
                  style={{
                    borderRadius: 12,
                    border: `1px solid ${border}`,
                    background: panel,
                    padding: 10,
                    minWidth: 0,
                  }}
                >
                  <div style={{ fontSize: 10, color: subtle, marginBottom: 6 }}>{m.label}</div>
                  <div style={{ fontSize: 13, whiteSpace: "nowrap", overflow: "hidden", textOverflow: "ellipsis" }}>
                    {m.value}
                  </div>
                </div>
              ))}
            </div>
          </>
        ) : null}

        <div style={{ height: 12 }} />

        <div style={{ display: "flex", justifyContent: "space-between", gap: 12, alignItems: "center" }}>
          <div style={{ fontSize: 11, color: subtle, whiteSpace: "nowrap", overflow: "hidden", textOverflow: "ellipsis" }}>
            {shareUrl ? shareUrl : watermark}
          </div>
          <div style={{ fontSize: 11, color: subtle }}>{watermark}</div>
        </div>
      </div>

      <div style={{ height: 12 }} />

      <div style={{ display: "flex", gap: 10, flexWrap: "wrap" }}>
        {shareUrl ? (
          <button
            onClick={onCopyLink}
            style={{
              padding: "10px 12px",
              borderRadius: 10,
              border: "1px solid rgba(0,0,0,0.15)",

            opacity: isDownloading ? 0.6 : 1,
          }}
        >
          {isDownloading ? "Downloading..." : "Download Image"}

            onClick={onNativeShare}
            style={{
              padding: "10px 12px",
              borderRadius: 10,
              border: "1px solid rgba(0,0,0,0.15)",
              background: "transparent",
              cursor: "pointer",
            }}
          >
            Share
          </button>
        ) : null}

    </div>
  );

        title={card.playerName}
        subtitle={card.title}
        takeText={takeText}
        metrics={metrics}
        shareUrl={shareUrl}
      />
    </div>
  );
}
