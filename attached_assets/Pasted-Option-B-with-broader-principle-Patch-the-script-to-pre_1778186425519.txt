typescriptconst TERMINAL_TIERS_NO_DOWNGRADE = new Set([
  "HOF",
  "LEGEND_DECEASED",
  "RETIRED"
]);

function isDowngrade(currentTier: string, newTier: string): boolean {
  if (!currentTier || currentTier === "") return false; // NULL fills always allowed
  if (currentTier === newTier) return false;
  
  // Preserve terminal tiers - don't downgrade them
  if (TERMINAL_TIERS_NO_DOWNGRADE.has(currentTier)) {
    // Allow upgrades within terminal tiers (LEGEND_DECEASED > HOF > RETIRED)
    // Block any other transition
    const isUpgradeWithinTerminal = (
      (currentTier === "RETIRED" && (newTier === "HOF" || newTier === "LEGEND_DECEASED")) ||
      (currentTier === "HOF" && newTier === "LEGEND_DECEASED")
    );
    return !isUpgradeWithinTerminal;
  }
  
  return false;
}
